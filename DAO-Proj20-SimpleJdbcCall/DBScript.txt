create or replace procedure P_AUTHENTICATION(user in varchar2,
                                             pass in varchar2,
                                             result out varchar2)
as
cnt number(2);
begin
  select count(*) into cnt from userlist where username=user and password=pass;
  if(cnt<>0) then
     result:='VALID CREDENTIALS';
  else
     result:='INVALID CREDENTIALS';
  end if;
end;
